001package asset.pipeline.grails.utils.net;
002
003
004import javax.servlet.http.HttpServletRequest;
005
006import static asset.pipeline.utils.net.Urls.getDefaultPort;
007import static asset.pipeline.utils.net.Urls.hasAuthority;
008import static org.codehaus.groovy.grails.web.util.WebUtils.getForwardURI;
009
010
011/**
012 * @author Ross Goldberg
013 */
014public final class HttpServletRequests {
015
016        public static String getScheme(final HttpServletRequest req) {
017                final String forwardedScheme = req.getHeader("x-forwarded-proto");
018                return
019                        forwardedScheme == null
020                                ? req.getScheme()
021                                : forwardedScheme
022                ;
023        }
024
025
026        public static int getPort(final HttpServletRequest req) {
027                final String forwardedPort = req.getHeader("x-forwarded-port");
028                return
029                        forwardedPort == null
030                                ? req.getServerPort()
031                                : Integer.parseInt(forwardedPort)
032                ;
033        }
034
035
036        public static StringBuilder getAuthorityUrl(final HttpServletRequest req, final Object o) {
037                return
038                        o == null
039                                ? getAuthorityUrlWithScheme(req)
040                                : o instanceof Boolean
041                                        ? getAuthorityUrl(req, ((Boolean) o).booleanValue())
042                                        : getAuthorityUrlWithScheme(req, o.toString())
043                ;
044        }
045
046        public static StringBuilder getAuthorityUrl(final HttpServletRequest req, final boolean retainScheme) {
047                return
048                        retainScheme
049                                ? getAuthorityUrlWithScheme(req)
050                                : getAuthorityUrlSansScheme(req)
051                ;
052        }
053
054        public static StringBuilder getAuthorityUrlWithScheme(final HttpServletRequest req) {
055                return getAuthorityUrlWithScheme(req, getScheme(req), true);
056        }
057
058        public static StringBuilder getAuthorityUrlWithScheme(final HttpServletRequest req, final String scheme) {
059                return getAuthorityUrlWithScheme(req, scheme, getScheme(req).equals(scheme));
060        }
061
062        public static StringBuilder getAuthorityUrlWithScheme(final HttpServletRequest req, final String scheme, final boolean includeNonDefaultPort) {
063                return getAuthorityUrlSansScheme(req, includeNonDefaultPort, new StringBuilder().append(scheme).append(':'));
064        }
065
066        public static StringBuilder getAuthorityUrlSansScheme(final HttpServletRequest req) {
067                return getAuthorityUrlSansScheme(req, false);
068        }
069
070        public static StringBuilder getAuthorityUrlSansScheme(final HttpServletRequest req, final String schemeToIncludeNonDefaultPort) {
071                return getAuthorityUrlSansScheme(req, getScheme(req).equals(schemeToIncludeNonDefaultPort));
072        }
073
074        public static StringBuilder getAuthorityUrlSansScheme(final HttpServletRequest req, final boolean includeNonDefaultPort) {
075                return getAuthorityUrlSansScheme(req, includeNonDefaultPort, new StringBuilder());
076        }
077
078        private static StringBuilder getAuthorityUrlSansScheme(final HttpServletRequest req, final boolean includeNonDefaultPort, final StringBuilder urlSb) {
079                urlSb.append("//").append(req.getServerName());
080
081                if (includeNonDefaultPort) {
082                        final int port = getPort(req);
083
084                        if (
085                                port >= 0 &&
086                                port != getDefaultPort(getScheme(req))
087                        ) {
088                                urlSb.append(':').append(port);
089                        }
090                }
091
092                return urlSb;
093        }
094
095

097                return
098                        o == null

100                                : o instanceof Boolean

104        }
105

107                return
108                        retainScheme
109                                ? getBaseUrlWithScheme(req)
110                                : getBaseUrlSansScheme(req)
111                ;
112        }
113
114        public static StringBuilder getBaseUrlWithScheme(final HttpServletRequest req) {
115                return getBaseUrlWithScheme(req, getScheme(req), true);
116        }
117
118        public static StringBuilder getBaseUrlWithScheme(final HttpServletRequest req, final String scheme) {
119                return getBaseUrlWithScheme(req, scheme, getScheme(req).equals(scheme));
120        }
121
122        public static StringBuilder getBaseUrlWithScheme(final HttpServletRequest req, final String scheme, final boolean includeNonDefaultPort) {
123                return getBaseUrlSansScheme(req, includeNonDefaultPort, new StringBuilder().append(scheme).append(':'));
124        }
125
126        public static StringBuilder getBaseUrlSansScheme(final HttpServletRequest req) {
127                return getBaseUrlSansScheme(req, false);
128        }
129
130        public static StringBuilder getBaseUrlSansScheme(final HttpServletRequest req, final String schemeToIncludeNonDefaultPort) {
131                return getBaseUrlSansScheme(req, getScheme(req).equals(schemeToIncludeNonDefaultPort));
132        }
133
134        public static StringBuilder getBaseUrlSansScheme(final HttpServletRequest req, final boolean includeNonDefaultPort) {
135                return getBaseUrlSansScheme(req, includeNonDefaultPort, new StringBuilder());
136        }
137
138        private static StringBuilder getBaseUrlSansScheme(final HttpServletRequest req, final boolean includeNonDefaultPort, final StringBuilder urlSb) {
139                return getAuthorityUrlSansScheme(req, includeNonDefaultPort, urlSb).append(req.getContextPath());
140        }
141
142
143        public static String prependBaseUrlWithScheme(final HttpServletRequest req, final String url) {
144                return
145                        hasAuthority(url)
146                                ? url
147                                : getBaseUrlWithScheme(req).append(url).toString()
148                ;
149        }

151

163                return
164                        retainScheme

168        }
169
170        public static StringBuilder getForwardUrlWithScheme(final HttpServletRequest req) {
171                return getForwardUrlWithScheme(req, getScheme(req), true);
172        }

187                return getForwardUrlSansScheme(req, getScheme(req).equals(schemeToIncludeNonDefaultPort));
188        }
189
190        public static StringBuilder getForwardUrlSansScheme(final HttpServletRequest req, final boolean includeNonDefaultPort) {
191                return getForwardUrlSansScheme(req, includeNonDefaultPort, new StringBuilder());
192        }
193
194        private static StringBuilder getForwardUrlSansScheme(final HttpServletRequest req, final boolean includeNonDefaultPort, final StringBuilder urlSb) {
195                return getAuthorityUrlSansScheme(req, includeNonDefaultPort, urlSb).append(getForwardURI(req));
196        }
197
198
199        private HttpServletRequests() {}
